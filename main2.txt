package main

import (
	"github.com/gin-gonic/gin"
)

func main() {
    router := gin.Default() //gin.Defaultは*gin.EngineをReturnする関数で、GenではこのEngineを使ってエンドポイントの追加やミドルウェアの登録を行うGinのコア的なやつ。 https://qiita.com/ozora/items/0597e52b3f9c1759e292#gindefault
    router.LoadHTMLGlob("templates/*.html")

    data := "Hello Go/Gin!!"

    router.GET("/", func(ctx *gin.Context){ //ここのGETをPOSTに変えることで、POSTを送ることもできる
        ctx.HTML(200, "index.html", gin.H{"data": data})
    })

    router.Run()
}
